CREATE TABLE leagues(
    id SERIAL PRIMARY KEY,
    league_ticker varchar(5) UNIQUE NOT NULL,
    league_full_name varchar(50) NOT NULL,
    league_hyperlink varchar(100) NOT NULL
)

CREATE TABLE teams(
    id SERIAL PRIMARY KEY,
    league_full_name varchar(50) NOT NULL REFERENCES leagues(league_full_name),
    league_ticker varchar(5) NOT NULL REFERENCES leagues(league_ticker),
    team_name varchar(80) NOT NULL,
    team_ticker varchar(5) NOT NULL
)

CREATE TABLE matches (
    id SERIAL PRIMARY KEY,
    id_flashscore_match varchar(20) UNIQUE NOT NULL,
    league_ticker varchar(5) NOT NULL REFERENCES leagues(league_ticker),
    match_date date NOT NULL,
    match_time time NOT NULL,
    home_team_ticker varchar(5) NOT NULL REFERENCES teams(team_ticker),
    away_team_ticker varchar(5) NOT NULL REFERENCES teams(team_ticker),
    hyperlink varchar(100) NOT NULL,
    ht_q1_pts int NOT NULL,
    at_q1_pts int NOT NULL,
    ht_q2_pts int NOT NULL,
    at_q2_pts int NOT NULL,
    ht_q3_pts int NOT NULL,
    at_q3_pts int NOT NULL,
    ht_q4_pts int NOT NULL,
    at_q4_pts int NOT NULL,
    ht_q5_pts int,
    at_q5_pts int,
    ht_odds_start,
    at_odds_start,
    ht_odds_end,
    at_odds_end,
)

CREATE TABLE players_info(
    id SERIAL PRIMARY KEY,
    id_flashscore_player varchar(20) UNIQUE NOT NULL,
    full_name varchar(50) NOT NULL,
    birth_date date NOT NULL,
    height int NOT NULL,
    hyperlink varchar(80) NOT NULL
    )

--all possible data stored in numeric format, null values should be trated as 0
CREATE TABLE players_statistic(
    id SERIAL PRIMARY KEY,
    id_match int REFERENCE matches(id),
    id_flashscore_match varchar(20) NOT NULL REFERENCES matches(id_flashscore_match),
    id_player int NOT NULL REFERENCES players_info(id),
    name_short varchar(20) NOT NULL, --short name from flashscore ex. Enoch S.
    team_ticker varchar(5) NOT NULL REFERENCES teams(team_ticker),
    pts int NOT NULL,
    reb int NOT NULL,
    ast int NOT NULL,
    min int NOT NULL,--time on field in seconds
    fgm int NOT NULL,
    fga int NOT NULL,
    2pm int NOT NULL,   
    2pa int NOT NULL,
    3pm int NOT NULL,   
    3pa int NOT NULL,
    ftm int NOT NULL,
    fta int NOT NULL,
    fta int NOT NULL,
    plus_minus int NOT NULL,
    or int NOT NULL,
    dr int NOT NULL,
    pf int NOT NULL,
    st int NOT NULL,
    to int NOT NULL,
    bs int NOT NULL,
    ba int NOT NULL,
    tfs int NOT NULL,
)

CRETE TABLE player_carrer(
    id SERIAL PRIMARY KEY,
    id_player int NOT NULL REFERENCES players_info(id),
    id_flashscore_player varchar(20) NOT NULL REFERENCES players_info(id_flashscore_player),
    season_start_year int NOT NULL,
    season_end_year int NOT NULL,
    team_ticker varchar(5) NOT NULL REFERENCES teams(team_ticker)
    league_ticker varchar(5) NOT NULL REFERENCES leagues(league_ticker)
)
